Code Preview: BFS Traversal

Level-order graph traversal:

void Graph::BFS(int start) {
    std::vector<bool> visited(numVertices, false);
    std::queue<int> q;

    visited[start] = true;
    q.push(start);

    std::cout << "BFS Traversal: ";
    while (!q.empty()) {
        int vertex = q.front();
        q.pop();
        std::cout << vertex << " ";

        // Visit all adjacent vertices
        for (int neighbor : adjList[vertex]) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                q.push(neighbor);
            }
        }
    }
    std::cout << std::endl;
}

// Dijkstra's Shortest Path (simplified)
void Graph::dijkstra(int start) {
    std::vector<int> dist(numVertices, INT_MAX);
    std::priority_queue<std::pair<int,int>,
        std::vector<std::pair<int,int>>,
        std::greater<std::pair<int,int>>> pq;

    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        for (auto& edge : adjList[u]) {
            int v = edge.first;
            int weight = edge.second;

            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pq.push({dist[v], v});
            }
        }
    }
}

Key insight: BFS uses queue (FIFO), DFS uses stack/recursion (LIFO). Dijkstra's uses priority queue for greedy shortest path. O(V + E) for BFS/DFS, O((V + E) log V) for Dijkstra's.

Drew's Tips for Success

  • BFS finds shortest path: In unweighted graphs, BFS naturally finds shortest path (fewest edges).
  • DFS explores deep: Great for detecting cycles, topological sort, pathfinding in mazes.
  • Dijkstra's needs weights: Shortest path in weighted graphs (maps with distances, network latency).
  • Adjacency list for sparse graphs: Most real-world graphs are sparse (social networks, web pages). Use adjacency list.
